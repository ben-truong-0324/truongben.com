Project Setup From your macOS terminal, create a new NUnit test project and add the Playwright dependencies: Bash # Create the test project dotnet new nunit -n WTS.WebApp.E2ETests # Add it to your solution dotnet sln add WTS.WebApp.E2ETests/WTS.WebApp.E2ETests.csproj # Add Playwright cd WTS.WebApp.E2ETests dotnet add package Microsoft.Playwright.NUnit # Build the project to download the Playwright binaries dotnet build # Install the Playwright browsers locally pwsh bin/Debug/net10.0/playwright.ps1 install 2. Homepage Features Test This test class targets the homepage (/) and validates three specific features based on the HTML our custom renderers output: the accordion navigation state, the dynamic grid column count, and the news layout structure. C# using Microsoft.Playwright; using Microsoft.Playwright.NUnit; using NUnit.Framework; using System.Threading.Tasks; namespace WTS.WebApp.E2ETests { [Parallelizable(ParallelScope.Self)] [TestFixture] public class HomepageTests : PageTest { // Adjust this to match your local Kestrel/IIS Express port private const string BaseUrl = "https://localhost:5001"; [SetUp] public async Task Setup() { await Page.GotoAsync(BaseUrl); } [Test] public async Task Feature1_AccordionNavigation_ExpandsAndUpdatesAriaState() { // Locate the first accordion button from the :::rcl-header-bar-nav var toggleButton = Page.Locator(".nav-accordion-toggle").First; // Verify initial state is closed await Expect(toggleButton).ToHaveAttributeAsync("aria-expanded", "false"); // Click the toggle await toggleButton.ClickAsync(); // Verify ARIA state updated and the submenu is now visible await Expect(toggleButton).ToHaveAttributeAsync("aria-expanded", "true"); var menuId = await toggleButton.GetAttributeAsync("aria-controls"); var subMenu = Page.Locator($"#{menuId}"); // Ensure the hidden attribute was removed by your vanilla JS await Expect(subMenu).Not.ToHaveAttributeAsync("hidden", ""); } [Test] public async Task Feature2_NewsLinksLayout_RendersTwoColumns() { // Verify the :::rcl-news-links generated the correct structural layout var leftColumn = Page.Locator(".col-lg-7"); var rightColumn = Page.Locator(".col-lg-5"); await Expect(leftColumn).ToBeVisibleAsync(); await Expect(rightColumn).ToBeVisibleAsync(); // Verify a specific news header exists in the left column var newsHeader = leftColumn.Locator("h2:has-text('News & Updates')"); await Expect(newsHeader).ToBeVisibleAsync(); } [Test] public async Task Feature3_DynamicRowGrid_AppliesCorrectColumnClasses() { // Assuming a :::rcl-row with 3 items was placed on the homepage // Verify the renderer applied the col-md-4 class dynamically var gridItems = Page.Locator(".row > .col-md-4"); // Ensure at least 3 items rendered with this class await Expect(gridItems).ToHaveCountAsync(3); } } } 3. About Page Test This validates navigation to the /about route and checks for expected content. C# using Microsoft.Playwright; using Microsoft.Playwright.NUnit; using NUnit.Framework; using System.Text.RegularExpressions; using System.Threading.Tasks; namespace WTS.WebApp.E2ETests { [Parallelizable(ParallelScope.Self)] [TestFixture] public class AboutPageTests : PageTest { private const string BaseUrl = "https://localhost:5001"; [Test] public async Task AboutPage_LoadsSuccessfully_AndDisplaysCorrectTitle() { // Navigate directly to the about page await Page.GotoAsync($"{BaseUrl}/about"); // Verify the page title matches await Expect(Page).ToHaveTitleAsync(new Regex("About Us - WTS WebApp")); // Verify the main H1 exists var mainHeading = Page.Locator("h1"); await Expect(mainHeading).ToContainTextAsync("About Our Platform"); } } }

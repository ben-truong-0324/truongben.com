The Core Parser Utility This utility uses a slightly modified Regex that safely reads across newlines inside the { ... } brackets. It completely separates the extraction logic from the HTML generation logic. C# using System.Collections.Generic; using System.Text.RegularExpressions; using WTS.RazorLibrary.Models; namespace WTS.RazorLibrary.Parsers { public static class RclParser { // Extracts the directive name, the optional attribute block (across multiple lines), and the content private static readonly Regex BlockRegex = new Regex( @":::(?[\w-]+)\s*(?:\{(?[^}]*)\})?\r?\n(?.*?)\r?\n:::", RegexOptions.Singleline | RegexOptions.Compiled); // Extracts key="value" pairs, ignoring newlines and spaces between them private static readonly Regex AttrRegex = new Regex( @"(?[\w-]+)\s*=\s*""(?[^""]*)""", RegexOptions.Compiled); /// /// Finds all ::: directives in the markdown and replaces them using the provided evaluator function. /// public static string ProcessDirectives(string markdown, string targetDirective, System.Func htmlEvaluator) { if (string.IsNullOrWhiteSpace(markdown)) return string.Empty; return BlockRegex.Replace(markdown, match => { var block = new RclBlock { DirectiveName = match.Groups["name"].Value.Trim(), Content = match.Groups["content"].Value.Trim() }; // Only process if it matches the specific directive we are looking for if (!string.Equals(block.DirectiveName, targetDirective, System.StringComparison.OrdinalIgnoreCase)) { return match.Value; // Leave untouched if it's a different directive } // Parse attributes if they exist var attrsRaw = match.Groups["attrs"].Value; if (!string.IsNullOrWhiteSpace(attrsRaw)) { foreach (Match attrMatch in AttrRegex.Matches(attrsRaw)) { block.Attributes[attrMatch.Groups["key"].Value] = attrMatch.Groups["value"].Value; } } // Hand the parsed block over to the renderer return htmlEvaluator(block); }); } } } 3.
